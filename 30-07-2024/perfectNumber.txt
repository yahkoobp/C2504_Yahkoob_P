pseudo : routine
function isPerfectNumber(num){
    set sum = 0
    temp = n
    for i from 1 to n-1 do
      if temp % i ==0
         sum = sum + i

    return sum == n
}

pseudo: routine test
function testIsPerfectNumber(){
    input num
    if isPerfectNumber(num)
      print("perfect number)
    else
      print("not perfect number")
    end if
}

pseudo: test routine in the driver code / main function 
function __Main__(){
    testIsPerfectNumber()
}
