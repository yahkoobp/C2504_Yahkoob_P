	DECLARE double total = 0;
	int num;
   	num = length(prices)
	for int i from 0 to num-1{
    		total = total + i
	if total > discountValue
        	discountedAmount = ApplyDiscount(discountPercentage , total)
		return discountedAmount
        else
                return total
        end if


function ApplyDiscount(float discountPercentage , float sellingPrice){
	double discount

4)Pseudocode to check whether a given string is palindrome or not

function isPalindrome(string){
    length = length(string)
    set last_index = length - 1
    set isPal = 1
    for i from 0 to length-1 / 2 do 
        if string[i] != string[last_index]
		set isPal = 0
                break
        last_index = last_index - 1
        end if
    end for

    return isPal

    }

function testisPalindrome(){
	input string str
        if isPalindrome(str)
		print("palindrome")
	else
		print("not palindrome")
}

5)Pseudocode to find largest element from an array

function largestFromArray(int array){
	int max = 0
	int length = length(array)
	for int i from 0 to length do
		if array[i] > max then
			set max = array[i]
		end if
        return max
}

function testLargestFromArray(){
	input int array
	int largest = largestFromArray(array)
        print(largest)
}

   testLargestFromArray()

	if(accountNumber is valid){
        	fetch total_amount
		set total_amount = total_amount + amount
		print("money deposited successfully")
		return total_amount
        else{
		print("Account is not valid")
                return -1
		}

}

}
